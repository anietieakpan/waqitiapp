Getting Started

Sign up for a developer account and generate your API keys from the dashboard.

Choose your preferred language and install our official SDK for quick integration.

3. Make First Call

Start with a simple payment request to test your integration.

Quick Start Example

curl -X POST https://api.example.com/v1/payments \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "type": "p2p",
    "amount": 5000,
    "currency": "USD",
    "recipient": {
      "type": "email",
      "value": "friend@example.com"
    },
    "description": "Lunch money"
  }'
import { WaqitiClient } from '@waqiti/sdk';

const client = new WaqitiClient('YOUR_API_KEY');

const payment = await client.payments.create({
  type: 'p2p',
  amount: 5000,
  currency: 'USD',
  recipient: {
    type: 'email',
    value: 'friend@example.com'
  },
  description: 'Lunch money'
});

console.log('Payment created:', payment.id);

import com.waqiti.sdk.WaqitiClient;
import com.waqiti.sdk.models.*;

WaqitiClient client = new WaqitiClient("YOUR_API_KEY");

CreatePaymentRequest request = CreatePaymentRequest.builder()
    .type(PaymentType.P2P)
    .amount(5000)
    .currency("USD")
    .recipient(Recipient.builder()
        .type(RecipientType.EMAIL)
        .value("friend@example.com")
        .build())
    .description("Lunch money")
    .build();

Payment payment = client.payments().create(request);
System.out.println("Payment created: " + payment.getId());
package main

import (
    "fmt"
    "github.com/waqiti/waqiti-go"
)

func main() {
    client := waqiti.NewClient("YOUR_API_KEY")
    
    payment, err := client.Payments.Create(&waqiti.PaymentParams{
        Type:     "p2p",
        Amount:   5000,
        Currency: "USD",
        Recipient: &waqiti.Recipient{
            Type:  "email",
            Value: "friend@example.com",
        },
        Description: "Lunch money",
    })
    
    if err != nil {
        panic(err)
    }
    
    fmt.Printf("Payment created: %s\n", payment.ID)
}

Authentication

All API requests require authentication using a Bearer token in the Authorization header:

Authorization: Bearer YOUR_API_KEY

Security Note: Always keep your API keys secure and never expose them in client-side code.

Webhooks

Webhooks allow you to receive real-time notifications about events in your Waqiti account.

Available Events

  • payment.created
  • payment.completed
  • payment.failed
  • payment.refunded
  • money_request.created
  • money_request.paid

Webhook Security

All webhooks are signed using HMAC-SHA256. Verify the signature to ensure the webhook is from Waqiti:

const crypto = require('crypto');

function verifyWebhook(payload, signature, secret) {
  const hmac = crypto.createHmac('sha256', secret);
  const digest = hmac.update(payload).digest('hex');
  return digest === signature;
}

Error Handling

The API uses standard HTTP status codes and returns detailed error objects:

{
  "error": {
    "code": "validation_error",
    "message": "The provided data failed validation",
    "field": "amount",
    "details": {
      "amount": "Amount must be greater than 0"
    },
    "requestId": "req_abc123"
  }
}

Common Error Codes

Status Code Description
400 invalid_request The request is missing required parameters
401 unauthorized Invalid or missing authentication token
403 forbidden You do not have permission to access this resource
404 not_found The requested resource was not found
422 validation_error The provided data failed validation
429 rate_limit_exceeded Too many requests
